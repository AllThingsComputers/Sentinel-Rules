 // Description of the Detection:   
// References (if available): https://resource.redcanary.com/rs/003-YRU-314/images/2020-Threat-Detection-Report.pdf
// The Logic of the Detection or Query Begins Below This Line
SysmonEvent
| where (EventID == 1 and Image matches regex @"(?i)\\(Rundll32|regasm|regsvr32|regsvcs|svchost|werfault)\.exe" and not("[-/]")) or (EventID == 1 and Image matches regex @"\\vbc\.exe" and "[-/\\]scomma|shtml|stext") or (EventID == 8 and endswith @"\Mshta.exe") or (EventID == 3 and Image endswith "Svchost.exe" and Protocol=~ "tcp" and DestinationPort matches regex @"447|449") or (EventID == 3 and Image endswith "notepad.exe" and not(ipv4_is_private(tostring(DestinationIp))))
