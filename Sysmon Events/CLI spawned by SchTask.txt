// Title of Detection:    CLI spawned by SchTask
// Authorship & Contact: https://allthingscomputers.medium.com  https://twitter.com/allthingshandle  https://allthingscomputersblog.wordpress.com/ 
 // Description of the Detection: 
// References (if available):   https://redcanary.com/blog/cryptomining-enabled-by-native-windows-tools/
// The Logic of the Detection or Query Begins Below This Line
SysmonEvent
| where EventID == 1 and ParentImage matches regex @"(?i)(taskeng|schtasks).exe" and Image matches regex @"(?i)(cmd|pwsh|powershell)\.exe$"
