// The Logic of the Detection or Query Begins Below This Line
SysmonEvent
| where (EventID == 1 and ParentImage matches regex @"(?i)\\Verclsid\.exe" and isnotempty(Image)) or EventID == 3 and Image matches regex @"(?i)\\Verclsid\.exe"
